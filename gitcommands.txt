git rm --cached <filename> = It will upstaged the file / It will remove files from staging area
git rm --cached <filename> -f = It staging area file and the upstaged file is diff and you still want to unstage you have to use the -f (Force) option
git restore

git commit = It will commit the file file and the editor will be opened where you should give the message and close it
git commit -m "First Commit / Initial Version" = It is used to commit the files to the git repository
-m = It is used to give message

git log = It will show the list of all commit logs

git diff = The command shows the file diff which are not yet staged and the last committed files
git diff --staged = The command shows the file diff which are in staged area and the last committed files

git show =It will show entire commit log for each file

git log --stat = It will display a list of files with addition and deletions
git log --pretty = It will format the log
git log --pretty=oneline = It will show the log in one line
git log --graph --pretty = It will show the log in graphical format
git log --pretty --format=short
git log --pretty --format=med = (by default)
git log --pretty --format=full
git log --pretty --format=fuller
git log --pretty --format=email
git log --pretty --format=raw

git reset <commit> = It will undo all the commits till the specified commit id and preserves the changes locally
git reset --hard <commit> = Command discards all history and goes back to specified commit, files locally will be overridden

git remote -v = It will list down all the remote repositories attached

git ls-files = It will list down all the file names which are committed

git push -u origin master = The data is pushed to remote repository from local repository
git pull origin master = The data is pulled from remote repository to local repository

git clone "<github project url>" = It will copy the entire github project on your machine
